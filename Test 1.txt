import sys
stdout = sys.stdout
rr = lambda: input().strip()
rri = lambda: int(rr())
rrm = lambda: list(map(int, rr().split()))

def solve(N, B, A):
    A.sort()
    ans = 0
    for x in A:
        if (B-x>=0):
            B=B-x
            ans+=1
        else:
            break
    return ans



T = rri()
for tc in range(1, T+1):
    N,P = rrm()
    A = rrm()
    ans = solve(N, P, A)
    print("Case #{}: {}".format(tc, ans))
