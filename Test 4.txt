import math
import sys
stdout = sys.stdout
rr = lambda: input().strip()
rri = lambda: int(rr())
rrm = lambda: list(map(int, rr().split()))

def solve(N,k, A):
    
    c=0
    for i in range(N - 1):
        c=max(c,A[i+1]-A[i])
    return int(round(c/k))

T = rri()
for tc in range(1, T+1):
    N,P = rrm()
    A = rrm()
    ans = solve(N, P,A)
    print("Case #{}: {}".format(tc, ans))
